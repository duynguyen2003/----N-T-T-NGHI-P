src/Backend/
├── config/              ← Cấu hình ứng dụng
│   ├── database.js      → Kết nối database, connection pooling
│   ├── jwt.js           → Cấu hình token JWT
│   ├── constants.js     → Các hằng số (roles, quyền, mã lỗi)
│   └── .env.example     → Template biến môi trường
│
├── middleware/          ← Kẻ "cổng vệ" xử lý request trước khi vào controller
│   ├── auth.js          → Xác thực token, kiểm tra quyền
│   ├── errorHandler.js  → Bắt lỗi chung từ toàn app
│   ├── validation.js    → Kiểm tra dữ liệu đầu vào
│   └── logging.js       → Ghi log request
│
├── routes/              ← Định nghĩa các endpoint API
│   ├── index.js         → Tổng hợp tất cả routes
│   ├── auth.js          → Routes đăng ký, đăng nhập
│   ├── courses.js       → Routes khóa học
│   ├── exams.js         → Routes kỳ thi
│   └── ... (6 routes khác)
│
├── controllers/         ← Xử lý HTTP requests (lấy dữ liệu, gọi service)
│   ├── authController.js
│   ├── courseController.js
│   └── ... (6 controllers khác)
│
├── services/            ← Chứa logic kinh doanh (phần "não" của app)
│   ├── authService.js   → Đăng ký, đăng nhập, token
│   ├── userService.js   → Quản lý user
│   ├── contentService.js → Quản lý course, lesson, module
│   └── ... (2 services khác)
│
├── validators/          ← Kiểm tra dữ liệu có hợp lệ không
│   ├── authValidator.js → Kiểm tra email, password
│   └── courseValidator.js
│
├── utils/               ← Các hàm tiện ích chung
│   ├── errors.js        → Định nghĩa lỗi (AuthError, NotFoundError...)
│   ├── response.js      → Format response API
│   └── helpers.js       → Hàm mã hóa, hash password, tạo token...
│
└── Server.js            ← File khởi động server (đã cải thiện)
